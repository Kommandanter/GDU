add_namespace = gduevents

#### Event that gives phoney war to allies ####
country_event = { 
	id = gduevents.1
	title = gduevents.1.t
	desc = gduevents.1.d
	picture = GFX_report_event_canadian_soldiers
	
	trigger = {
		has_global_flag = war_is_hell 
		NOT = {
			has_country_flag = war_is_hell_2
		}
		OR = { 
			tag = IRE 
			tag = HOL
			tag = ENG
			is_in_faction_with = ENG 
		}
		NOT = { tag = FRA }
		NOT = { tag = USA }
	}
		
	
	mean_time_to_happen = {
		days = 2
	}
	
	immediate = { 
		set_country_flag = war_is_hell_2
	}
	
	option = {
		name = gduevents.1.a
		trigger = { NOT = { tag = FRA } } 
		add_timed_idea = { idea = alli_phoney_war days = 240 }
		add_ideas = war_economy
	
	}
	
	option = {
		name = gduevents.1.a
		trigger = { tag = FRA }
		add_timed_idea = { idea = FRA_phoney_war days = 240 }
	
	}
	
	
}
