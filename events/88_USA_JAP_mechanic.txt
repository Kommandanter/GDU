
add_namespace = usajap_events

country_event = {
	id = usajap_events.1
    hidden = yes

	is_triggered_only = yes
	
	immediate = { 

    set_variable = {
		usa_naval_factories = num_of_naval_factories
	}
    hidden_effect = { JAP = { country_event = { id = usajap_events.2 days = 15 } } }
    }	
}
country_event = {
	id = usajap_events.2
    hidden = yes

	is_triggered_only = yes
    
    immediate = { 
    set_variable = {
		
		jap_naval_factories = num_of_naval_factories
	}

    divide_variable = { USA.usa_naval_factories = jap_naval_factories }
        if = {
				limit = { check_variable = { USA.usa_naval_factories < 0.66 } }
                JAP = { add_ideas = USA_low_dockyard_count }
				
		}
	
	else = {
	
	JAP = { remove_ideas = USA_low_dockyard_count }
	}	
    
    hidden_effect = { USA = { country_event = { id = usajap_events.1 days = 15 } }
    }
    }
    	
}
